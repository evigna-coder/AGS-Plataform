Create `apps/sistema-modular/src/pages/stock/hooks/useEquivalenciaListExpansion.ts`:

```ts
import { useCallback, useMemo } from 'react';
import type { Articulo } from '@ags/shared';

interface Params {
  articulos: Articulo[];
  searchTerm: string;
}

interface Result {
  destinoLookup: Map<string, Articulo>;
  hasEquivalencia: (a: Articulo) => boolean;
  shouldExpandRow: (a: Articulo) => boolean;
}

/**
 * Phase 13 STKE-07 — hook owning the equivalencia expansion logic for ArticulosList.
 *
 * Extracted unconditionally from ArticulosList.tsx (NOT a "if-LOC-budget-crosses" extraction)
 * to keep the list page rendering pure and the expansion logic testable in isolation.
 *
 * - destinoLookup: O(1) reverse map from destino articuloId → origen Articulo (built from articulos[])
 * - hasEquivalencia: true when the article has either side of a 1→1 link
 * - shouldExpandRow: true only when the user searched exactly for the codigo of either side of a linked pair
 */
export function useEquivalenciaListExpansion({ articulos, searchTerm }: Params): Result {
  const destinoLookup = useMemo<Map<string, Articulo>>(() => {
    const m = new Map<string, Articulo>();
    for (const a of articulos) {
      if (a.articuloIdDestinoEquivalencia) m.set(a.articuloIdDestinoEquivalencia, a);
    }
    return m;
  }, [articulos]);

  const hasEquivalencia = useCallback((a: Articulo) => {
    return !!a.equivalencias?.length || destinoLookup.has(a.id);
  }, [destinoLookup]);

  const shouldExpandRow = useCallback((a: Articulo) => {
    const q = searchTerm.trim().toLowerCase();
    if (!q) return false;
    // Direct match on this article's codigo
    if (a.codigo.toLowerCase() === q) {
      return hasEquivalencia(a);
    }
    // Indirect match on its partner's codigo
    const linkedCode =
      a.equivalencias?.[0]?.articuloCodigoDestino?.toLowerCase() ??
      destinoLookup.get(a.id)?.codigo.toLowerCase();
    return !!linkedCode && linkedCode === q;
  }, [searchTerm, destinoLookup, hasEquivalencia]);

  return { destinoLookup, hasEquivalencia, shouldExpandRow };
}
```

 * existing e2e helpers (see firestore-assert.ts for the pattern). Returns the seeded

export async function seedEquivalenciaPair(opts: {
  codigoOrigen?: string;
  codigoDestino?: string;
  factor?: number;
} = {}): Promise<{ origenId: string; destinoId: string; codigoOrigen: string; codigoDestino: string; factor: number }> {
  const codigoOrigen = opts.codigoOrigen ?? `TEST-COMPRA-${Date.now()}`;
  const codigoDestino = opts.codigoDestino ?? `TEST-USO-${Date.now()}`;
  const factor = opts.factor ?? 10;

  // 1. Create the destino artículo first (origen needs to denormalize its codigo/descripcion)
  const destinoRef = adminDb.collection('articulos').doc();
  const destinoId = destinoRef.id;
  await destinoRef.set({
    id: destinoId,
    codigo: codigoDestino,
    descripcion: `E2E destino ${codigoDestino}`,
    // minimum required fields per Articulo shape (confirm by reading packages/shared)
    categoriaEquipo: 'consumibles',
    marcaId: 'test-marca',
    proveedorIds: [],
    tipo: 'consumible',
    unidadMedida: 'unidad',
    stockMinimo: 0,
    activo: true,
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
    equivalencias: [],
    articuloIdDestinoEquivalencia: null,
  });

  // 2. Create the origen with the equivalencia + flat field pointed at the destino
  const origenRef = adminDb.collection('articulos').doc();
  const origenId = origenRef.id;
  await origenRef.set({
    id: origenId,
    codigo: codigoOrigen,
    descripcion: `E2E origen ${codigoOrigen}`,
    categoriaEquipo: 'consumibles',
    marcaId: 'test-marca',
    proveedorIds: [],
    tipo: 'consumible',
    unidadMedida: 'unidad',
    stockMinimo: 0,
    activo: true,
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
    equivalencias: [{
      articuloIdDestino: destinoId,
      articuloCodigoDestino: codigoDestino,
      articuloDescripcionDestino: `E2E destino ${codigoDestino}`,
      factor,
    }],
    articuloIdDestinoEquivalencia: destinoId,
  });

  return { origenId, destinoId, codigoOrigen, codigoDestino, factor };
}

```typescript
import { db, deepCleanForFirestore, getUpdateTrace } from './firebase';
import { doc, getDoc, setDoc, onSnapshot } from 'firebase/firestore';
import type { AdminConfigFlujos } from '@ags/shared';

const DOC_PATH = 'adminConfig/flujos';

export const ADMIN_CONFIG_DEFAULTS: Pick<AdminConfigFlujos, 'mailFacturacion'> = {
  mailFacturacion: 'mbarrios@agsanalitica.com',
};

export const adminConfigService = {
  async get(): Promise<AdminConfigFlujos | null> {
    const snap = await getDoc(doc(db, DOC_PATH));
    return snap.exists() ? ({ ...(snap.data() as any) } as AdminConfigFlujos) : null;
  },
  async getWithDefaults(): Promise<AdminConfigFlujos> {
    const cfg = await this.get();
    return cfg || ({ ...ADMIN_CONFIG_DEFAULTS, updatedAt: new Date().toISOString() } as AdminConfigFlujos);
  },
  async update(data: Partial<AdminConfigFlujos>, updatedBy: string, updatedByName?: string): Promise<void> {
    await setDoc(doc(db, DOC_PATH), deepCleanForFirestore({
      ...data,
      updatedAt: new Date().toISOString(),
      updatedBy,
      updatedByName: updatedByName ?? null,
    }), { merge: true });
  },
  subscribe(callback: (cfg: AdminConfigFlujos | null) => void, onError?: (e: Error) => void): () => void {
    return onSnapshot(doc(db, DOC_PATH), (snap) => {
      callback(snap.exists() ? (snap.data() as AdminConfigFlujos) : null);
    }, onError);
  },
};
```

**Regla 1 — `arrayUnion` prohibido dentro de `runTransaction`:**
```typescript
// ❌ BAD — arrayUnion no es transaccional
tx.update(presRef, { ordenesCompraIds: arrayUnion(ocId) });

// ✅ GOOD — merge manual tras tx.get
const presSnap = await tx.get(presRef);
const current = (presSnap.data()?.ordenesCompraIds as string[] | undefined) || [];
if (!current.includes(ocId)) {
  tx.update(presRef, { ordenesCompraIds: [...current, ocId], updatedAt: Timestamp.now() });
}
```

**Regla 2 — no llamar otros services que abran batch/tx:**
```typescript
// ❌ BAD — leadsService.update usa batch → nested tx problem
await runTransaction(db, async (tx) => {
  await leadsService.update(leadId, { estado: 'oc_recibida' }); // NO
});

// ✅ GOOD — inline de writes
await runTransaction(db, async (tx) => {
  const leadSnap = await tx.get(doc(db, 'leads', leadId));
  if (!leadSnap.exists()) throw new Error('Ticket no encontrado');
  const lead = leadSnap.data() as Lead;
  tx.update(leadSnap.ref, deepCleanForFirestore({
    estado: 'oc_recibida',
    postas: [...(lead.postas || []), nuevaPosta],
    updatedAt: Timestamp.now(),
  }));
});
```

**Regla 3 — todas las reads ANTES de todas las writes.**

**Regla 4 — side effect de `cargarOC`:**
Dado `{ payload, context: { leadId, presupuestosIds, existingOcId } }`:
1. Read: todos los presupuestos en `presupuestosIds` + lead (si leadId) + OC existente (si existingOcId) + articulos referenciados por items de los presupuestos (solo si necesitamos recomputar `itemRequiereImportacion` — en Phase 8 usamos el flag precomputado en el item, NO reads de articulos dentro de tx — ver regla H del RESEARCH).
2. Validaciones: presupuestos `estado === 'aceptado'` (si no, arrojar error para no cargar OC sobre borradores); cliente coincide en todos los presupuestos si es OC compartida.
3. Writes:
   - Crear o updatear `ordenesCompraCliente/{ocId}` con `presupuestosIds` merged
   - Update cada `presupuesto.ordenesCompraIds` con merge manual
   - Update cada `lead.estado = 'oc_recibida'` + append Posta con `estadoAnterior` del lead actual
   - NO appendear pendingAction `'derivar_comex'` en `cargarOC` (W1 fix 2026-04-21): la derivación a Comex ya ocurrió al `aceptar` (plan 08-04 `aceptarConRequerimientos`). Appendearla acá crearía un orphan que el retry handler de 08-03 trata como no-op.
   - NO appendear pendingAction `'notificar_coordinador_ot'` dentro de la tx principal. Es un side-effect post-commit — se registra SOLO si la notificación real falla (ver paso 4).
4. Post-commit (fuera de tx, async, best-effort): intentar notificar al coordinador OT usando `adminConfig/flujos.usuarioCoordinadorOTId` (append Posta en el ticket + optionally side-channel). Envolver en try/catch: **si la notificación falla** (ej. `usuarioCoordinadorOTId` no configurado, usuario inactivo, error de side-channel) → llamar `presupuestosService._appendPendingAction(pid, { type: 'notificar_coordinador_ot', reason: <err msg concreto> })`. Si el side-effect tiene éxito, NO se registra nada. Esto alinea el dashboard: solo aparecen pendientes reales.

```typescript
async cargarOC(
  payload: Omit<OrdenCompraCliente, 'id' | 'createdAt' | 'updatedAt'>,
  context: { leadId?: string | null; presupuestosIds: string[]; existingOcId?: string | null },
  actor?: { uid: string; name?: string }
): Promise<{ id: string; numero: string }> {
  if (!context.presupuestosIds.length) {
    throw new Error('cargarOC: al menos 1 presupuesto requerido');
  }

  const nowIso = new Date().toISOString();
  const ocRef = context.existingOcId
    ? doc(db, 'ordenesCompraCliente', context.existingOcId)
    : doc(collection(db, 'ordenesCompraCliente'));
  const finalOcId = ocRef.id;

  await runTransaction(db, async (tx) => {
    // ── READS (todas primero) ───────────────────────────────────────
    const presSnaps = await Promise.all(
      context.presupuestosIds.map((id) => tx.get(doc(db, 'presupuestos', id)))
    );
    const leadSnap = context.leadId ? await tx.get(doc(db, 'leads', context.leadId)) : null;
    const ocSnap = context.existingOcId ? await tx.get(ocRef) : null;

    // ── VALIDATIONS ─────────────────────────────────────────────────
    const presupuestos = presSnaps.map((s, i) => {
      if (!s.exists()) throw new Error(`Presupuesto ${context.presupuestosIds[i]} no encontrado`);
      const p = { id: s.id, ...(s.data() as any) } as Presupuesto;
      if (p.estado !== 'aceptado') {
        throw new Error(`Presupuesto ${p.numero || p.id} no está aceptado (estado actual: ${p.estado}). No se puede cargar OC.`);
      }
      return p;
    });

    // ── WRITES ──────────────────────────────────────────────────────
    if (context.existingOcId && ocSnap?.exists()) {
      const existing = ocSnap.data() as OrdenCompraCliente;
      const mergedPresIds = Array.from(new Set([...(existing.presupuestosIds || []), ...context.presupuestosIds]));
      const mergedAdjuntos = [...(existing.adjuntos || []), ...(payload.adjuntos || [])];
      tx.update(ocRef, deepCleanForFirestore({
        presupuestosIds: mergedPresIds,
        adjuntos: mergedAdjuntos,
        notas: payload.notas ?? existing.notas,
        updatedAt: nowIso,
        updatedBy: actor?.uid ?? null,
        updatedByName: actor?.name ?? null,
      }));
    } else {
      tx.set(ocRef, deepCleanForFirestore({
        numero: payload.numero,
        fecha: payload.fecha,
        clienteId: payload.clienteId,
        presupuestosIds: context.presupuestosIds,
        adjuntos: payload.adjuntos || [],
        notas: payload.notas ?? null,
        createdAt: nowIso,
        updatedAt: nowIso,
        createdBy: actor?.uid ?? null,
        createdByName: actor?.name ?? null,
      }));
    }

    // Per-presupuesto: update ordenesCompraIds + append pendingAction
    for (const p of presupuestos) {
      const currentOcIds = p.ordenesCompraIds || [];
      const newOcIds = currentOcIds.includes(finalOcId) ? currentOcIds : [...currentOcIds, finalOcId];

      // W1 fix: NO appendear pendingAction dentro de la tx. derivar_comex ya ocurrió en acceptance (08-04);
      // notificar_coordinador_ot se evalúa post-commit (try/catch del side-effect real).
      tx.update(doc(db, 'presupuestos', p.id), deepCleanForFirestore({
        ordenesCompraIds: newOcIds,
        updatedAt: nowIso,
      }));
    }

    // Lead update + posta append
    if (leadSnap?.exists()) {
      const lead = leadSnap.data() as Lead;
      const nuevaPosta: Posta = {
        id: crypto.randomUUID(),
        fecha: nowIso,
        estadoAnterior: lead.estado,
        estadoNuevo: 'oc_recibida',
        deUsuarioId: actor?.uid ?? null,
        deUsuarioNombre: actor?.name ?? null,
        motivo: `OC ${payload.numero} cargada para presupuesto(s) ${presupuestos.map(p => p.numero).join(', ')}`,
      } as any; // shape final queda per Posta existing — ajustar al tipo real
      const nuevasPostas = [...(lead.postas || []), nuevaPosta];
      tx.update(leadSnap.ref, deepCleanForFirestore({
        estado: 'oc_recibida' as TicketEstado,
        postas: nuevasPostas,
        updatedAt: nowIso,
      }));
    }
  });

  // ── W1 fix: post-commit side-effect best-effort — notificar coordinador OT ───
  // Si falla, appendeamos pendingAction 'notificar_coordinador_ot' con reason concreto.
  try {
    const cfg = await adminConfigService.getWithDefaults();
    if (!cfg.usuarioCoordinadorOTId) {
      throw new Error('adminConfig/flujos.usuarioCoordinadorOTId no configurado');
    }
    // TODO: side-channel notification (in-app posta / toast admin / mail) — v2.0 best-effort.
    // Minimum: verify usuario activo; si no activo, throw → pendingAction
    // (usuario lookup via usuariosService.getById si está disponible)
  } catch (err: any) {
    for (const pid of context.presupuestosIds) {
      try {
        await presupuestosService._appendPendingAction(pid, {
          type: 'notificar_coordinador_ot',
          reason: err?.message || 'notificación coordinador OT falló',
        });
      } catch (appendErr) {
        console.error('[cargarOC] _appendPendingAction failed:', appendErr);
      }
    }
  }

  return { id: finalOcId, numero: payload.numero };
},
```

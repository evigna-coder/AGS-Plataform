**`otService.cerrarAdministrativamente` contract:**

```typescript
/**
 * Cierra una OT administrativamente en una sola transacción atómica.
 * - Update OT: estadoAdmin='CIERRE_ADMINISTRATIVO' + fechaCierre
 * - Crear ticket interno al área 'administracion'
 * - Encolar doc en 'mailQueue' con type='cierre_admin_ot'
 *
 * Post-commit:
 * - syncFromOT en el lead origen (si existe) — best-effort
 * - Si el encolado falla → pendingAction en los presupuestos vinculados
 */
async cerrarAdministrativamente(
  otNumber: string,
  cierreData: { notas?: string; fechaCierre?: string },
  actor?: { uid: string; name?: string }
): Promise<{ adminTicketId: string; mailQueueId: string }>;
```

**Shape del doc en `mailQueue`:**

```typescript
{
  type: 'cierre_admin_ot',
  status: 'pending',
  data: {
    to: string,                    // mailFacturacion de adminConfig o default
    subject: `Aviso facturación — OT ${otNumber}`,
    body: string,                  // template con cliente, presupuesto, OCs, OTs
    otNumber: string,
    presupuestoIds: string[],      // OT.budgets
    ocIds: string[],               // derived: flat from all presupuestos.ordenesCompraIds
    attachments: [                 // metadata; el Cloud Function consumer resuelve URLs
      { type: 'pdf_presupuesto', presupuestoId },
      { type: 'oc_adjunto', ocId, adjuntoId },
      // OT PDFs si existen
    ],
  },
  createdAt: string,
  createdBy?: string,
}
```

**Regla:** NO llamar a `otService.update` / `leadsService.create` / `presupuestosService.update` dentro de la tx. Inline writes.

**`cerrarAdministrativamente` — detalle de reads/writes:**

Reads:
- `otRef` — para validar existencia y leer `budgets[]`, `leadId`
- Opcional: presupuestos vinculados (si se necesitan para el body del mail — leer FUERA de tx para no inflar lecturas)

Writes dentro de tx:
- `tx.update(otRef, { estadoAdmin: 'CIERRE_ADMINISTRATIVO', fechaCierre: nowIso, updatedAt: nowIso })`
- `tx.set(newAdminTicketRef, { motivoContacto: 'Aviso facturación...', estado: 'nuevo', areaActual: 'administracion', ... })`
- `tx.set(newMailQueueRef, { type: 'cierre_admin_ot', status: 'pending', data: {...}, createdAt })`

Post-commit (fuera de tx):
- `leadsService.syncFromOT(ot.leadId, otNumber, 'CIERRE_ADMINISTRATIVO')` si aplica
- NO-OP adicional — la tx ya se completó; side effect best-effort

**Error paths:**
- Si el render del body falla (ej. presupuesto no existe) → fallback body mínimo y pendingAction reason='body_render_fallback'
- Si la tx falla completa → error bubbles up a `useOTDetail` → toast

```ts
async cerrarAdministrativamente(
  otNumber: string,
  cierreData: { notas?: string; fechaCierre?: string },
  actor?: { uid: string; name?: string }
): Promise<{ adminTicketId: string; mailQueueId: string }> {
  // Pre-reads fuera de tx
  const ot = await this.getByOtNumber(otNumber);
  if (!ot) throw new Error('OT no encontrada');
  const cfg = await adminConfigService.getWithDefaults();
  const mailTo = cfg.mailFacturacion || 'mbarrios@agsanalitica.com';

  // Armar metadata del mail FUERA de tx (si falla, throw antes de tx)
  const presupuestoIds = ot.budgets || [];
  const presupuestos = await Promise.all(presupuestoIds.map((pid) => presupuestosService.getById(pid)));
  const ocIds = Array.from(new Set(presupuestos.flatMap((p) => (p?.ordenesCompraIds || []))));
  const subject = `Aviso facturación — OT ${otNumber}`;
  const body = buildAvisoFacturacionBody(ot, presupuestos); // helper separado

  const nowIso = new Date().toISOString();
  const newAdminTicketRef = doc(collection(db, 'leads'));
  const newMailQueueRef = doc(collection(db, 'mailQueue'));

  await runTransaction(db, async (tx) => {
    const otRef = doc(db, 'ordenesTrabajo', otNumber);
    const otSnap = await tx.get(otRef);
    if (!otSnap.exists()) throw new Error('OT no encontrada (tx)');

    // Update OT
    tx.update(otRef, deepCleanForFirestore({
      estadoAdmin: 'CIERRE_ADMINISTRATIVO',
      fechaCierreAdmin: cierreData.fechaCierre ?? nowIso,
      notasCierre: cierreData.notas ?? null,
      updatedAt: nowIso,
      updatedBy: actor?.uid ?? null,
      updatedByName: actor?.name ?? null,
    }));

    // Ticket admin nuevo
    tx.set(newAdminTicketRef, deepCleanForFirestore({
      motivoContacto: `Aviso facturación — OT ${otNumber}`,
      descripcion: `OT ${otNumber} cerrada administrativamente. Presupuesto(s) ${presupuestoIds.join(', ')}. Revisar facturación.`,
      estado: 'nuevo' as TicketEstado,
      areaActual: 'administracion' as TicketArea,
      clienteId: ot.clienteId ?? null,
      origen: { tipo: 'ot', id: otNumber, ref: otNumber },
      createdAt: nowIso,
      createdBy: actor?.uid ?? null,
      createdByName: actor?.name ?? null,
    }));

    // mailQueue doc
    tx.set(newMailQueueRef, deepCleanForFirestore({
      type: 'cierre_admin_ot',
      status: 'pending',
      data: {
        to: mailTo,
        subject,
        body,
        otNumber,
        presupuestoIds,
        ocIds,
        attachments: [
          ...presupuestoIds.map((pid) => ({ type: 'pdf_presupuesto', presupuestoId: pid })),
          // OC attachments se resuelven en el consumer via ocIds
        ],
      },
      createdAt: nowIso,
      createdBy: actor?.uid ?? null,
    }));
  });

  // Post-commit side effects (best-effort)
  try {
    if (ot.leadId) {
      await leadsService.syncFromOT(ot.leadId, otNumber, 'CIERRE_ADMINISTRATIVO');
    }
  } catch (err) {
    console.error('[cerrarAdministrativamente] syncFromOT failed:', err);
  }

  return { adminTicketId: newAdminTicketRef.id, mailQueueId: newMailQueueRef.id };
},
```

Helper `buildAvisoFacturacionBody`:

```ts
function buildAvisoFacturacionBody(ot: WorkOrder, presupuestos: (Presupuesto | null)[]): string {
  const lines = [
    `OT ${ot.otNumber} cerrada administrativamente.`,
    ``,
    `Cliente: ${ot.clienteNombre || ot.clienteId}`,
    `Fecha de cierre: ${new Date().toLocaleDateString('es-AR')}`,
    ``,
    `Presupuestos vinculados:`,
    ...presupuestos.filter(Boolean).map((p) => `  - ${p!.numero} (${p!.moneda}) — total ${p!.total}`),
    ``,
    `Órdenes de compra vinculadas:`,
    // ...flatMap OCs; el consumer del mailQueue resuelve numeros reales
    `  (ver adjuntos)`,
    ``,
    `Adjunto: PDF del presupuesto + PDFs de OC`,
  ];
  return lines.join('\n');
}
```

```tsx
import { useEffect, useState } from 'react';
import { Button, Input, SearchableSelect, Card } from '../../components/ui';
import { adminConfigService } from '../../services/adminConfigService';
import { usuariosService } from '../../services/usuariosService';
import { useAuth } from '../../contexts/AuthContext';
import { toast } from 'sonner';
import type { AdminConfigFlujos, Usuario } from '@ags/shared';

export default function ConfigFlujosPage() {
  const { firebaseUser, usuario } = useAuth();
  const [form, setForm] = useState<Partial<AdminConfigFlujos>>({ mailFacturacion: '' });
  const [usuarios, setUsuarios] = useState<Usuario[]>([]);
  const [loading, setLoading] = useState(true);
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    (async () => {
      const [cfg, allUsuarios] = await Promise.all([
        adminConfigService.getWithDefaults(),
        usuariosService.getAll({ activo: true }),
      ]);
      setForm(cfg);
      setUsuarios(allUsuarios);
      setLoading(false);
    })();
  }, []);

  const handleSave = async () => {
    setSubmitting(true);
    setError(null);
    try {
      // Validar activos
      for (const [field, userId] of [
        ['seguimiento', form.usuarioSeguimientoId],
        ['coordinador OT', form.usuarioCoordinadorOTId],
        ['responsable Comex', form.usuarioResponsableComexId],
      ] as const) {
        if (userId) {
          const u = usuarios.find((x) => x.id === userId);
          if (!u || u.activo === false) {
            throw new Error(`Usuario ${field} no está activo`);
          }
        }
      }
      if (form.mailFacturacion && !form.mailFacturacion.includes('@')) {
        throw new Error('Mail facturación inválido');
      }
      await adminConfigService.update(form, firebaseUser!.uid, usuario?.nombre);
      toast.success('Configuración guardada');
    } catch (err: any) {
      setError(err.message);
    } finally {
      setSubmitting(false);
    }
  };

  if (loading) return <div className="p-8">Cargando...</div>;

  return (
    <div className="max-w-2xl p-8 space-y-6">
      <h1 className="font-serif text-3xl text-slate-900">Config Flujos</h1>
      <Card className="space-y-4 p-6">
        <div>
          <label className="block text-[10px] font-mono uppercase tracking-wide text-slate-600 mb-1">
            Seguimiento comercial (FLOW-01)
          </label>
          <SearchableSelect
            value={form.usuarioSeguimientoId || ''}
            onChange={(v) => setForm({ ...form, usuarioSeguimientoId: v })}
            options={usuarios.map((u) => ({ value: u.id, label: `${u.nombre} (${u.email})` }))}
          />
        </div>
        <div>
          <label className="block text-[10px] font-mono uppercase tracking-wide text-slate-600 mb-1">
            Coordinador OT (FLOW-02)
          </label>
          <SearchableSelect
            value={form.usuarioCoordinadorOTId || ''}
            onChange={(v) => setForm({ ...form, usuarioCoordinadorOTId: v })}
            options={usuarios.map((u) => ({ value: u.id, label: `${u.nombre} (${u.email})` }))}
          />
        </div>
        <div>
          <label className="block text-[10px] font-mono uppercase tracking-wide text-slate-600 mb-1">
            Responsable Comex (opcional, FLOW-03)
          </label>
          <SearchableSelect
            value={form.usuarioResponsableComexId || ''}
            onChange={(v) => setForm({ ...form, usuarioResponsableComexId: v })}
            options={[{ value: '', label: '(Sin usuario fijo — deriva por área)' }, ...usuarios.map((u) => ({ value: u.id, label: `${u.nombre} (${u.email})` }))]}
          />
        </div>
        <div>
          <label className="block text-[10px] font-mono uppercase tracking-wide text-slate-600 mb-1">
            Mail facturación (FLOW-04)

            value={form.mailFacturacion || ''}

            placeholder="mbarrios@agsanalitica.com"
          />
        </div>
        {error && <div className="text-sm text-red-700 bg-red-50 p-3 rounded">{error}</div>}
        <Button onClick={handleSave} disabled={submitting} className="w-full">
          {submitting ? 'Guardando…' : 'Guardar'}
        </Button>
      </Card>
    </div>
  );
}
```

```tsx
import { useEffect, useMemo, useState } from 'react';
import { Link } from 'react-router-dom';
import { Button, Card, SearchableSelect } from '../../components/ui';
import { useUrlFilters } from '../../hooks/useUrlFilters';
import { presupuestosService } from '../../services/presupuestosService';
import { useAuth } from '../../contexts/AuthContext';
import { toast } from 'sonner';
import type { Presupuesto, PendingAction } from '@ags/shared';

const FILTERS_SCHEMA = {
  tipo: { type: 'string' as const, default: '' },
  antiguedad: { type: 'string' as const, default: '' },
  clienteId: { type: 'string' as const, default: '' },
};

type Row = {
  presupuestoId: string;
  presupuestoNumero: string;
  clienteId: string;
  action: PendingAction;
};

export default function AccionesPendientesPage() {
  const [filters, setFilter, , reset] = useUrlFilters(FILTERS_SCHEMA);
  const { firebaseUser, usuario } = useAuth();
  const [rows, setRows] = useState<Row[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Preferido: subscribe. Para simplicidad inicial, fetch + filtrado in-memory
    const unsub = presupuestosService.subscribeAll((presupuestos: Presupuesto[]) => {
      const collected: Row[] = [];
      for (const p of presupuestos) {
        for (const a of (p.pendingActions || [])) {
          if (!a.resolvedAt) {
            collected.push({
              presupuestoId: p.id,
              presupuestoNumero: p.numero,
              clienteId: p.clienteId,
              action: a,
            });
          }
        }
      }
      setRows(collected);
      setLoading(false);
    });
    return () => { try { unsub?.(); } catch {} };
  }, []);

  const filtered = useMemo(() => {
    return rows.filter((r) => {
      if (filters.tipo && r.action.type !== filters.tipo) return false;
      if (filters.clienteId && r.clienteId !== filters.clienteId) return false;
      if (filters.antiguedad) {
        const ageMs = Date.now() - new Date(r.action.createdAt).getTime();
        const ageDays = ageMs / (1000 * 60 * 60 * 24);
        if (filters.antiguedad === 'nuevo' && ageDays > 1) return false;
        if (filters.antiguedad === 'mediana' && (ageDays < 1 || ageDays > 7)) return false;
        if (filters.antiguedad === 'vieja' && ageDays < 7) return false;
      }
      return true;
    });
  }, [rows, filters]);

  const actor = { uid: firebaseUser?.uid || '', name: usuario?.nombre };

  const handleRetry = async (row: Row) => {
    const result = await presupuestosService.retryPendingAction(row.presupuestoId, row.action.id, actor);
    if (result.success) toast.success(`Reintento OK — ${row.action.type}`);
    else toast.error(`Reintento falló: ${result.error}`);
  };

  const handleResolve = async (row: Row) => {
    if (!confirm('Marcar como resuelta manualmente?')) return;
    await presupuestosService.markPendingActionResolved(row.presupuestoId, row.action.id, actor);
    toast.success('Marcada resuelta');
  };

  if (loading) return <div className="p-8">Cargando…</div>;

  return (
    <div className="p-8 space-y-6">
      <h1 className="font-serif text-3xl text-slate-900">Acciones Pendientes</h1>
      {/* Filtros */}
      <Card className="p-4 flex gap-4 flex-wrap">
        {/* 3 selects: tipo, antiguedad, clienteId */}
      </Card>
      {filtered.length === 0 ? (
        <div className="text-center py-12 text-slate-500">Sin acciones pendientes 🎉</div>
      ) : (
        <table className="w-full">
          {/* thead + tbody renderizando filtered */}
        </table>
      )}
    </div>
  );
}
```

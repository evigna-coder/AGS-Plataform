export function itemRequiresImportacion(articulo: Articulo | null): boolean {
  if (!articulo) return false;

  const disponible = r.disponible || 0;

**Behavior detail:**

1. Read: `tx.get(presupuesto)`. Validar estado actual (debe permitir transición a aceptado; si ya está aceptado, no-op).
2. Filtrar `items` con `itemRequiereImportacion === true`. Para cada uno construir payload de Requerimiento con:
   - `estado: 'pendiente'`
   - `condicional: true`
   - `presupuestoId: presupuestoId`
   - `presupuestoNumero: pres.numero`
   - `articuloId: item.stockArticuloId`
   - `articuloDescripcion: item.descripcion`
   - `cantidad: item.cantidad`
   - `motivo: 'Auto — items sin stock en presupuesto aceptado'`
   - `origen: 'presupuesto_aceptado'` (o el enum que corresponda)
3. Writes (todas en tx):
   - Por cada requerimiento: `tx.set(newReqRef, payload)` con numero generado ANTES de la tx via `requerimientosService.getNextNumero()` (si esa función usa su propio runTransaction, cachear N numeros antes de iniciar la tx — simpler: reservar un numero por requerimiento fuera de la tx, luego set dentro).
   - Update del presupuesto: `estado: 'aceptado'`, `fechaAceptacion: now` (si el tipo lo tiene), `updatedAt: now`.
4. Fuera de la tx (post-commit side effect best-effort):
   - `leadsService.syncFromPresupuesto(origenId, numero, 'aceptado')` si aplica — reusar pattern existente
   - Posta en el ticket con `aArea: 'materiales_comex'` (o equivalente) derivando; si falla, registrar pendingAction 'derivar_comex' vía `this._appendPendingAction` (helper definido en plan 08-03 Task 1 Edit 3)

**Edit A — `presupuestosService.ts`: agregar `aceptarConRequerimientos`:**

```ts
async aceptarConRequerimientos(
  presupuestoId: string,
  actor?: { uid: string; name?: string }
): Promise<{ requerimientosIds: string[] }> {
  // Paso 1: pre-reservar numeros de requerimiento FUERA de tx
  // (requerimientosService.getNextNumero usa su propio runTransaction; no se puede invocar dentro)
  const pres = await this.getById(presupuestoId);
  if (!pres) throw new Error('Presupuesto no encontrado');
  const itemsImport = (pres.items || []).filter((it: any) => it.itemRequiereImportacion === true && it.stockArticuloId);

  const numerosReservados: string[] = [];
  for (const _ of itemsImport) {
    const numero = await requerimientosService.getNextNumero();
    numerosReservados.push(numero);
  }

  // Paso 2: runTransaction
  const nowIso = new Date().toISOString();
  const newReqIds: string[] = [];

  await runTransaction(db, async (tx) => {
    const presRef = doc(db, 'presupuestos', presupuestoId);
    const presSnap = await tx.get(presRef);
    if (!presSnap.exists()) throw new Error('Presupuesto no encontrado');
    const pp = presSnap.data() as Presupuesto;
    if (pp.estado === 'aceptado') {
      // idempotente: no-op
      return;
    }

    // Crear requerimientos (tx.set)
    itemsImport.forEach((item: any, idx: number) => {
      const reqRef = doc(collection(db, 'requerimientosCompra'));
      newReqIds.push(reqRef.id);
      tx.set(reqRef, deepCleanForFirestore({
        numero: numerosReservados[idx],
        articuloId: item.stockArticuloId,
        articuloCodigo: item.articuloCodigo ?? null,
        articuloDescripcion: item.descripcion,
        cantidad: item.cantidad,
        unidadMedida: item.unidadMedida || 'unidad',
        motivo: 'Auto — items sin stock en presupuesto aceptado',
        origen: 'presupuesto_aceptado',
        origenRef: presupuestoId,
        estado: 'pendiente',
        condicional: true,
        presupuestoId,
        presupuestoNumero: pp.numero,
        solicitadoPor: actor?.uid ?? null,
        fechaSolicitud: nowIso,
        createdAt: nowIso,
        updatedAt: nowIso,
        createdBy: actor?.uid ?? null,
        createdByName: actor?.name ?? null,
      }));
    });

    // Update del presupuesto
    tx.update(presRef, deepCleanForFirestore({
      estado: 'aceptado',
      updatedAt: nowIso,
      updatedBy: actor?.uid ?? null,
      updatedByName: actor?.name ?? null,
    }));
  });

  // Paso 3: post-commit side effects (fuera de tx)
  if (pres.origenTipo === 'lead' && pres.origenId) {
    try {
      await leadsService.syncFromPresupuesto(pres.origenId, pres.numero, 'aceptado');
    } catch (err) {
      console.error('[aceptarConRequerimientos] syncFromPresupuesto failed:', err);
    }
  }
  if (itemsImport.length > 0 && pres.origenTipo === 'lead' && pres.origenId) {
    // Posta al área materiales_comex — best-effort; si falla registra pendingAction
    // Nota: _appendPendingAction es definido por plan 08-03 Task 1 Edit 3. Dependencia hard (wave 3).
    try {
      await leadsService.derivarArea(pres.origenId, 'materiales_comex' as TicketArea, {
        motivo: `Auto — ${itemsImport.length} items requieren importación`,
      });
    } catch (err: any) {
      await this._appendPendingAction(presupuestoId, {
        type: 'derivar_comex',
        reason: `derivación a materiales_comex falló: ${err?.message || 'unknown'}`,
      });
    }
  }

  return { requerimientosIds: newReqIds };
},
```

**Edit B — `presupuestosService.ts`: agregar `_cancelarRequerimientosCondicionales`:**

```ts
async _cancelarRequerimientosCondicionales(
  presupuestoId: string,
  actor?: { uid: string; name?: string }
): Promise<{ cancelled: number; skipped: number }> {
  const reqs = await requerimientosService.getByPresupuesto(presupuestoId);
  const condicionales = reqs.filter((r) => r.condicional === true);
  const cancellables = condicionales.filter((r) => r.estado === 'pendiente' || r.estado === 'aprobado');
  const skipped = condicionales.length - cancellables.length;
  const nowIso = new Date().toISOString();

  const batch = createBatch();
  for (const r of cancellables) {
    batch.update(doc(db, 'requerimientosCompra', r.id), deepCleanForFirestore({
      estado: 'cancelado',
      canceladoPor: 'presupuesto_anulado',
      updatedAt: nowIso,
      updatedBy: actor?.uid ?? null,
      updatedByName: actor?.name ?? null,
    }));
  }
  await batch.commit();
  return { cancelled: cancellables.length, skipped };
},
```

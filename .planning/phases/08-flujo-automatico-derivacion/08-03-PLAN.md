```ts
async retryPendingAction(
  presupuestoId: string,
  actionId: string,
  actor?: { uid: string; name?: string }
): Promise<{ success: boolean; error?: string }> {
  const pres = await this.getById(presupuestoId);
  if (!pres) return { success: false, error: 'Presupuesto no encontrado' };
  const action = (pres.pendingActions || []).find((a) => a.id === actionId);
  if (!action) return { success: false, error: 'Action no encontrada' };
  if (action.resolvedAt) return { success: true };

  let success = false;
  let error: string | undefined;
  try {
    switch (action.type) {
      case 'crear_ticket_seguimiento':
        await this._crearAutoTicketSeguimiento(pres, actor);
        success = true;
        break;
      case 'derivar_comex':
        // plan 08-04 extiende este handler para crear requerimiento condicional tardío si llegó la OC antes
        // v2.0 fallback: success=true con warning (no-op — el requerimiento debe existir ya si FLOW-03 corrió)
        success = true;
        break;
      case 'notificar_coordinador_ot':
        // v2.0: posta al coordinador (implementado como side-effect best-effort en 08-04 o acá)
        // No bloquear v2.0: success=true; UI sabe que es "marcar OK" básicamente
        success = true;
        break;
      case 'enviar_mail_facturacion':
        // plan 08-05 extiende este handler con retry client-side del mail
        return { success: false, error: 'retry desde /admin/acciones-pendientes botón específico (plan 08-05)' };
      default:
        return { success: false, error: `tipo no soportado: ${action.type}` };
    }
  } catch (err: any) {
    error = err?.message || 'Error desconocido';
  }

  // Update: attempts++ siempre; resolvedAt solo si success
  const updated = (pres.pendingActions || []).map((a) => {
    if (a.id !== actionId) return a;
    return {
      ...a,
      attempts: (a.attempts || 0) + 1,
      ...(success ? { resolvedAt: new Date().toISOString() } : {}),
    };
  });
  await updateDoc(doc(db, 'presupuestos', presupuestoId), deepCleanForFirestore({
    pendingActions: updated,
    updatedAt: new Date().toISOString(),
  }));

  return success ? { success: true } : { success: false, error };
},

async retryPendingActionsForCliente(
  clienteId: string,
  actor?: { uid: string; name?: string }
): Promise<{ retried: number; successful: number; failed: number }> {
  const presupuestos = await this.getByCliente(clienteId); // usa método existente o crea query
  let retried = 0;
  let successful = 0;
  let failed = 0;
  for (const pres of presupuestos) {
    const pendientes = (pres.pendingActions || []).filter((a) => !a.resolvedAt);
    for (const action of pendientes) {
      retried++;
      const result = await this.retryPendingAction(pres.id, action.id, actor);
      if (result.success) successful++;
      else failed++;
    }
  }
  return { retried, successful, failed };
},

async markPendingActionResolved(
  presupuestoId: string,
  actionId: string,
  actor?: { uid: string; name?: string }
): Promise<void> {
  const pres = await this.getById(presupuestoId);
  if (!pres) return;
  const updated = (pres.pendingActions || []).map((a) =>
    a.id === actionId ? { ...a, resolvedAt: new Date().toISOString() } : a
  );
  await updateDoc(doc(db, 'presupuestos', presupuestoId), deepCleanForFirestore({
    pendingActions: updated,
    updatedAt: new Date().toISOString(),
  }));
},
```

```tsx
import React from 'react';
import type { Disponibilidad } from '@ags/shared';
import { DISPONIBILIDAD_LABELS } from '@ags/shared';

interface Props {
  disponibilidad: Disponibilidad | null | undefined;
  etaDiasEstimados: number | null | undefined;
  onChange: (next: { disponibilidad: Disponibilidad | null; etaDiasEstimados: number | null }) => void;
  variant?: 'modal' | 'row';
  atpHint?: { atp: number } | null;
  disabled?: boolean;
}

const DISP_OPTIONS: Array<{ value: Disponibilidad | ''; label: string }> = [
  { value: '',                 label: '— Sin definir —' },
  { value: 'stock',            label: DISPONIBILIDAD_LABELS.stock },
  { value: 'post_facturacion', label: DISPONIBILIDAD_LABELS.post_facturacion },
  { value: 'a_importar',       label: DISPONIBILIDAD_LABELS.a_importar },
  { value: 'en_transito',      label: DISPONIBILIDAD_LABELS.en_transito },
];

const LABEL_CLASS = 'block text-[10px] uppercase tracking-wide font-mono text-slate-500 mb-1';
const INPUT_CLASS = 'w-full text-xs border border-slate-200 rounded-lg px-2 py-1.5 bg-white text-slate-700 focus:outline-none focus:ring-2 focus:ring-teal-500 disabled:bg-slate-50 disabled:text-slate-400';

export const PresupuestoDisponibilidadFields: React.FC<Props> = ({
  disponibilidad, etaDiasEstimados, onChange, variant = 'modal', atpHint, disabled,
}) => {
  const handleDispChange = (v: string) => {
    const next = (v || null) as Disponibilidad | null;
    onChange({ disponibilidad: next, etaDiasEstimados: etaDiasEstimados ?? null });
  };
  const handleEtaChange = (v: string) => {
    const n = v.trim() === '' ? null : Number(v);
    onChange({ disponibilidad: disponibilidad ?? null, etaDiasEstimados: Number.isFinite(n as number) ? (n as number) : null });
  };

  const layoutCls = variant === 'row'
    ? 'grid grid-cols-2 gap-2'
    : 'space-y-3';

  return (
    <div className={layoutCls}>
      <div>
        <label className={LABEL_CLASS}>Disponibilidad</label>
        <select
          value={disponibilidad ?? ''}
          onChange={(e) => handleDispChange(e.target.value)}
          className={INPUT_CLASS}
          disabled={disabled}
          data-testid="disp-select"
        >
          {DISP_OPTIONS.map(o => (
            <option key={o.value || 'none'} value={o.value}>{o.label}</option>
          ))}
        </select>
        {atpHint && (
          <span className="block text-[10px] text-slate-400 mt-1 font-mono">
            ATP: {atpHint.atp}
          </span>
        )}
      </div>
      <div>
        <label className={LABEL_CLASS}>ETA (días desde aceptación)</label>
        <input
          type="number"
          inputMode="numeric"
          min={0}
          value={etaDiasEstimados ?? ''}
          onChange={(e) => handleEtaChange(e.target.value)}
          placeholder={variant === 'row' ? '—' : 'Ej: 30'}
          className={INPUT_CLASS}
          disabled={disabled}
          data-testid="disp-eta"
        />
      </div>
    </div>
  );
};
```

```typescript
useEffect(() => {
  // Si el operador ya seteó manualmente, no pisar.
  if (disponibilidadTouched) return;

  // Caso 1: item de servicio (sin articuloId de stock) → post_facturacion
  if (!stockArticuloId) {
    setDisponibilidad('post_facturacion');
    setEtaDiasEstimados(null);
    setAtpHint(null);
    return;
  }

  // Caso 2: item de stock → consultar ATP
  let cancelled = false;
  (async () => {
    try {
      const stockAmplio = await computeStockAmplio(stockArticuloId);
      if (cancelled) return;
      const atp = stockAmplio.atp ?? 0;
      setAtpHint({ atp });
      if (atp > 0) {
        setDisponibilidad('stock');
        setEtaDiasEstimados(0);
      } else {
        setDisponibilidad('a_importar');
        setEtaDiasEstimados(30);
      }
    } catch (err) {
      console.warn('[AddItemModal] computeStockAmplio failed; default to a_importar', err);
      if (!cancelled) {
        setDisponibilidad('a_importar');
        setEtaDiasEstimados(30);
        setAtpHint(null);
      }
    }
  })();
  return () => { cancelled = true; };
}, [stockArticuloId, disponibilidadTouched]);
```

```tsx
import React, { useState } from 'react';
import type { Disponibilidad } from '@ags/shared';
import { PresupuestoDisponibilidadFields } from './PresupuestoDisponibilidadFields';

interface Props {
  itemsCount: number;
  onApplyAll: (next: { disponibilidad: Disponibilidad | null; etaDiasEstimados: number | null }) => void;
  disabled?: boolean;
}

export const BulkAplicarDisponibilidadButton: React.FC<Props> = ({ itemsCount, onApplyAll, disabled }) => {
  const [open, setOpen] = useState(false);
  const [disp, setDisp] = useState<Disponibilidad | null>(null);
  const [eta, setEta] = useState<number | null>(null);

  if (itemsCount === 0) return null;

  const handleApply = () => {
    onApplyAll({ disponibilidad: disp, etaDiasEstimados: eta });
    setOpen(false);
  };

  return (
    <>
      <button
        type="button"
        onClick={() => setOpen(true)}
        disabled={disabled}
        className="text-[11px] font-medium text-teal-700 hover:text-teal-900 px-2 py-1 rounded hover:bg-teal-50 disabled:text-slate-400 disabled:cursor-not-allowed"
        data-testid="bulk-aplicar-disp"
      >
        Aplicar a todos los items ({itemsCount})
      </button>

      {open && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/40" onClick={() => setOpen(false)}>
          <div
            className="bg-white rounded-xl shadow-xl max-w-sm w-full p-4 mx-4"
            onClick={(e) => e.stopPropagation()}
          >
            <h3 className="text-sm font-semibold text-slate-700 mb-3" style={{ fontFamily: 'Newsreader, serif' }}>
              Aplicar a todos los items
            </h3>
            <p className="text-[11px] text-slate-500 mb-3">
              Estos valores se aplicarán a los {itemsCount} items del presupuesto.
              Se pueden ajustar individualmente después.
            </p>
            <PresupuestoDisponibilidadFields
              disponibilidad={disp}
              etaDiasEstimados={eta}
              onChange={(n) => { setDisp(n.disponibilidad); setEta(n.etaDiasEstimados); }}
              variant="modal"
            />
            <div className="flex justify-end gap-2 mt-4">
              <button
                type="button"
                onClick={() => setOpen(false)}
                className="text-xs px-3 py-1.5 rounded-lg border border-slate-200 text-slate-600 hover:bg-slate-50"
              >Cancelar</button>
              <button
                type="button"
                onClick={handleApply}
                className="text-xs px-3 py-1.5 rounded-lg bg-teal-700 text-white hover:bg-teal-800"
              >Aplicar</button>
            </div>
          </div>
        </div>
      )}
    </>
  );
};
```

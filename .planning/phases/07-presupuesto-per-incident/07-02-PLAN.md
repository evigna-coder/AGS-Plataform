<objective>
Implementar token-first order en envío por mail: validar/obtener OAuth token ANTES de cambiar estado en Firestore. Si el token falla o `sendGmail` falla, el presupuesto NO transiciona a `enviado`. Al éxito, hacer un único `updateDoc` atómico con `estado: 'enviado' + fechaEnvio: today`.

Purpose: Cierra Pitfall 5-A documentado en STATE.md. Hoy el flujo es: el usuario clickea "Enviar email" → `EnviarPresupuestoModal.handleSend()` corre `requestToken → generatePDF → sendGmail → onSent()` → y el callback `onSent` en `EditPresupuestoModal` (líneas 322-330) es quien llama `handleEstadoChange('enviado')` + `save()`. Si el token falla, `handleSend` throwea antes de llamar `onSent`, OK — pero hay otros bugs: (a) el `save()` dentro de `onSent` recorre TODO el form y re-serializa ítems/cuotas/etc, no es un update quirúrgico — si algo del form está en estado inconsistente, puede fallar post-envío y dejar el doc en `borrador` con el mail ya despachado; (b) el cambio de estado está desacoplado del resultado real del envío — si por alguna razón `onSent` no se invoca (ej. el component unmountea antes), el estado nunca cambia; (c) no hay manejo diferenciado de errores por etapa, todos muestran el mismo `error` genérico.

Output: Flow de envío atómico y confiable, reutilizable para `'partes'`, `'mixto'`, `'ventas'` de Phase 10.

**Step 1 — Crear el hook `useEnviarPresupuesto.ts` (MANDATORIO, no condicional):**

```tsx
// apps/sistema-modular/src/hooks/useEnviarPresupuesto.ts
import { useState } from 'react';
import { useGoogleOAuth } from './useGoogleOAuth';
import { sendGmail } from '../services/gmailService';
import { presupuestosService } from '../services/firebaseService';
import type { PresupuestoEstado } from '@ags/shared';
import type { GeneratePDFParams } from '../components/presupuestos/pdf/generatePresupuestoPDF';

export type EnviarStatus = 'idle' | 'authorizing' | 'generating_pdf' | 'sending' | 'updating_firestore' | 'sent' | 'error';

interface UseEnviarParams {
  presupuestoId: string;
  presupuestoEstado: PresupuestoEstado;
  presupuestoNumero: string;
  pdfParams: GeneratePDFParams;
  origenTipo?: string | null;
  origenId?: string | null;
  onSuccess: () => void;
}

export function useEnviarPresupuesto(params: UseEnviarParams) {
  const { requestToken } = useGoogleOAuth();
  const [status, setStatus] = useState<EnviarStatus>('idle');
  const [error, setError] = useState<string>('');
  const [sending, setSending] = useState(false);

  const send = async (opts: { to: string[]; cc?: string[]; subject: string; htmlBody: string }) => {
    setSending(true);
    setError('');

    // STAGE 1: OAuth token
    let accessToken: string;
    try {
      setStatus('authorizing');
      accessToken = await requestToken();
    } catch (err) {
      setError(`No se pudo autorizar Gmail: ${err instanceof Error ? err.message : 'Error desconocido'}. Podés reintentar.`);
      setStatus('error'); setSending(false);
      return;
    }

    // STAGE 2: Generate PDF
    let pdfBase64: string;
    try {
      setStatus('generating_pdf');
      const { generatePresupuestoPDF } = await import('../components/presupuestos/pdf');
      const pdfBlob = await generatePresupuestoPDF(params.pdfParams);
      pdfBase64 = await blobToBase64(pdfBlob);
    } catch (err) {
      setError(`No se pudo generar el PDF: ${err instanceof Error ? err.message : 'Error desconocido'}. Podés reintentar.`);
      setStatus('error'); setSending(false);
      return;
    }

    // STAGE 3: Send Gmail
    try {
      setStatus('sending');
      await sendGmail({
        accessToken,
        to: opts.to,
        cc: opts.cc && opts.cc.length > 0 ? opts.cc : undefined,
        subject: opts.subject,
        htmlBody: opts.htmlBody,
        attachments: [{
          filename: `${params.presupuestoNumero}.pdf`,
          mimeType: 'application/pdf',
          base64Data: pdfBase64,
        }],
      });
    } catch (err) {
      setError(`El mail no se pudo enviar: ${err instanceof Error ? err.message : 'Error desconocido'}. El estado NO cambió — podés reintentar.`);
      setStatus('error'); setSending(false);
      return;
    }

    // STAGE 4: Mark as enviado (atomic) — only if currently 'borrador', pass hint to skip getById
    if (params.presupuestoEstado === 'borrador') {
      try {
        setStatus('updating_firestore');
        await presupuestosService.markEnviado(params.presupuestoId, {
          origenTipo: params.origenTipo || undefined,
          origenId: params.origenId || undefined,
        });
      } catch (err) {
        setError(`Mail enviado OK, pero hubo un error actualizando el estado: ${err instanceof Error ? err.message : 'Error desconocido'}. El presupuesto sigue en "borrador" — cambialo manualmente.`);
        setStatus('error'); setSending(false);
        return;
      }
    }

    // STAGE 5: Success
    setStatus('sent');
    setTimeout(() => {
      params.onSuccess();
    }, 1500);
    setSending(false);
  };

  return { send, status, error, sending };
}

async function blobToBase64(blob: Blob): Promise<string> {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onloadend = () => {
      const result = reader.result as string;
      resolve(result.split(',')[1] || '');
    };
    reader.onerror = reject;
    reader.readAsDataURL(blob);
  });
}
```

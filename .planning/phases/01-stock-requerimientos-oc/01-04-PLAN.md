const aggregated = useMemo(() => {
  const byArticulo = new Map<string, { disponible: number; reservado: number; asignado: number; nombre: string; codigo: string }>();
  unidades.forEach(u => {

  return [...byArticulo.entries()].map(([articuloId, counts]) => ({

  }));

**Part A — Create ReservarStockModal component (multi-item aware):**

Create `apps/sistema-modular/src/components/stock/ReservarStockModal.tsx` (under 130 lines). This modal handles ALL stock-linked items in a presupuesto. The user first selects which item to reserve, then picks an available unit for that item.

```typescript
// ReservarStockModal props — accepts the full list of stock-linked items:
interface StockItem {
  articuloId: string;
  descripcion: string;
}

interface Props {
  presupuestoId: string;
  presupuestoNumero: string;
  clienteId: string;
  clienteNombre: string;
  items: StockItem[];           // ALL items with stockArticuloId from the presupuesto
  onClose: () => void;
  onSuccess: () => void;
}
```

The modal has two steps driven by `selectedItem: StockItem | null` state:

**Step 1 — Item selector (when `selectedItem === null`):**
- Shows a list or `<select>` of all items passed in `props.items`
- Each row: `{item.descripcion}` — clicking selects it and moves to Step 2
- Header: "Reservar Stock — Seleccionar artículo"
- If `items.length === 1`, auto-select it immediately (skip straight to Step 2)

**Step 2 — Unit selector (when `selectedItem` is set):**
- On mount of this step (useEffect on `selectedItem`): fetches `unidadesService.getAll({ articuloId: selectedItem.articuloId, estado: 'disponible' })`
- Shows a list of available units (nroSerie or `Unidad #${index+1}`, ubicacion.referenciaNombre)
- Admin clicks a unit row → calls `useReservaStock().reservar()` → on success calls `onSuccess()` + closes
- "← Volver" button to go back to Step 1 (set `selectedItem = null`)
- Header: `"Reservar Stock — ${selectedItem.descripcion}"`
- If no units available: show "No hay unidades disponibles para este artículo." with a "Volver" button

Use the standard Modal wrapper from `components/ui/Modal` (teal header, #FAFAFA body).
Header title (Newsreader): derived from the current step as above.
Both steps are within the same component — no separate components needed.

```typescript
import { useState, useEffect } from 'react';
import { unidadesService } from '../../services/stockService';
import { useReservaStock } from '../../hooks/useReservaStock';
import { Modal } from '../ui/Modal';
import type { UnidadStock } from '@ags/shared';

export function ReservarStockModal(props: Props) {
  const [selectedItem, setSelectedItem] = useState<StockItem | null>(
    props.items.length === 1 ? props.items[0] : null
  );
  const [unidades, setUnidades] = useState<UnidadStock[]>([]);
  const [loadingUnidades, setLoadingUnidades] = useState(false);
  const { reservar, loading: reservando } = useReservaStock();

  useEffect(() => {
    if (!selectedItem) return;
    setLoadingUnidades(true);
    unidadesService.getAll({ articuloId: selectedItem.articuloId, estado: 'disponible' })
      .then(setUnidades)
      .catch(() => setUnidades([]))
      .finally(() => setLoadingUnidades(false));
  }, [selectedItem]);

  const handleReservar = async (unidad: UnidadStock) => {
    const ok = await reservar({
      unidadId: unidad.id,
      unidad,
      presupuestoId: props.presupuestoId,
      presupuestoNumero: props.presupuestoNumero,
      clienteId: props.clienteId,
      clienteNombre: props.clienteNombre,
      solicitadoPorNombre: 'Admin',
    });
    if (ok) props.onSuccess();
  };

  // ... render: item selector or unit selector depending on selectedItem
}
```

const aggregated = useMemo(() => {
  const byArticulo = new Map<string, { disponible: number; reservado: number; asignado: number; nombre: string; codigo: string }>();
  unidades.forEach(u => {
    const prev = byArticulo.get(u.articuloId) ?? { disponible: 0, reservado: 0, asignado: 0, nombre: u.articuloDescripcion, codigo: u.articuloCodigo };
    if (u.estado === 'disponible') prev.disponible++;
    else if (u.estado === 'reservado') prev.reservado++;
    else if (u.estado === 'asignado') prev.asignado++;
    byArticulo.set(u.articuloId, prev);
  });
  return [...byArticulo.entries()].map(([articuloId, counts]) => ({
    articuloId, ...counts, total: counts.disponible + counts.reservado + counts.asignado
  }));
}, [unidades]);

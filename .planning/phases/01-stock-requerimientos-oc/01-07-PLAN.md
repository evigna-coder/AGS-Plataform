```typescript
const handleReponer = useCallback(async (
  item: InventarioItem,
  cantidad: number,
  depotPosicionId: string,
  depotPosicionNombre: string,
): Promise<boolean> => {
  if (cantidad <= 0) {
    console.warn('[handleReponer] cantidad must be > 0');
    return false;
  }
  try {
    // Create a MovimientoStock: transferencia from depot to minikit
    await movimientosService.create({
      tipo: 'transferencia',
      articuloId: item.articuloId,
      articuloCodigo: item.articuloCodigo ?? null,
      articuloDescripcion: item.articuloDescripcion ?? null,
      unidadStockId: null,  // quantity-based movement, no specific unit
      nroSerie: null,
      origenTipo: 'posicion',
      origenId: depotPosicionId,
      origenNombre: depotPosicionNombre,
      destinoTipo: item.ubicacion?.tipo ?? 'minikit',
      destinoId: item.ubicacion?.referenciaId ?? '',
      destinoNombre: item.ubicacion?.referenciaNombre ?? 'Minikit',
      cantidad,
      motivo: `Reposición de minikit: ${cantidad} unidades de ${item.articuloCodigo ?? item.articuloDescripcion}`,
      referenciaTipo: null,
      referenciaId: null,
      solicitadoPor: 'Admin',
      notas: null,
    });
    return true;
  } catch (err) {
    console.error('[useInventarioIngeniero] handleReponer:', err);
    return false;
  }
}, [movimientosService]);
```

```typescript
import { movimientosService } from '../../services/stockService';

const handleSubmit = async () => {
  if (!justificacion.trim()) {
    setJustifError('La justificación es obligatoria.');
    return;
  }
  if (delta === 0) {
    setJustifError('El ajuste no puede ser cero.');
    return;
  }
  setSaving(true);
  try {
    await movimientosService.create({
      tipo: 'ajuste',
      articuloId: props.unidad.articuloId,
      articuloCodigo: props.unidad.articuloCodigo,
      articuloDescripcion: props.unidad.articuloDescripcion,
      unidadStockId: props.unidad.id,
      nroSerie: props.unidad.nroSerie ?? null,
      origenTipo: props.unidad.ubicacion.tipo,
      origenId: props.unidad.ubicacion.referenciaId,
      origenNombre: props.unidad.ubicacion.referenciaNombre,
      destinoTipo: null,
      destinoId: null,
      destinoNombre: null,
      cantidad: delta,  // can be negative
      motivo: justificacion.trim(),
      notas: notas.trim() || null,
      referenciaTipo: null,
      referenciaId: null,
      solicitadoPor: 'Admin',
    });
    props.onSuccess();
  } catch (err) {
    setJustifError('Error al guardar el ajuste. Intente nuevamente.');
    console.error('[AjusteStockModal]', err);
  } finally {
    setSaving(false);
  }
};
```

    ```typescript
    async togglePreEmbarque(
      presupuestoId: string,
      next: boolean,
      actor?: { uid: string; name?: string },
    ): Promise<void> {
      const pres = await this.getById(presupuestoId);
      if (!pres) throw new Error('Presupuesto no encontrado');
      if (pres.estado === 'finalizado' || pres.estado === 'anulado') {
        throw new Error('No se puede modificar pre-embarque en un presupuesto cerrado');
      }
      // Idempotent — no-op if state matches
      if ((pres.preEmbarque ?? false) === next) return;

      // Update ppto
      await this.update(presupuestoId, deepCleanForFirestore({ preEmbarque: next }) as any);

      // Audit posta on linked ticket (best-effort — failure must not block toggle).
      // B4 fix: leadsService.appendPosta does NOT exist. Use the exact inline
      // write pattern from presupuestosService.ts:1199-1201 (contrato-OC posta):
      //   await leadsService.update(ticket.id, { postas: [...(ticket.postas || []), postaOC] });
      try {
        if (pres.leadId) {
          const { leadsService } = await import('./leadsService');
          const lead = await leadsService.getById(pres.leadId);
          if (lead) {
            const posta = deepCleanForFirestore({
              fecha: new Date().toISOString(),
              usuarioId: actor?.uid ?? null,
              usuarioNombre: actor?.name ?? null,
              accion: next ? 'pre_embarque_marcada' : 'pre_embarque_desmarcada',
              detalle: `Presupuesto N° ${pres.numero}: pre-embarque ${next ? 'sí' : 'no'}`,
            });
            await leadsService.update(pres.leadId, {
              postas: [...(lead.postas || []), posta],
            });
          }
        }
      } catch (err) {
        console.warn('[togglePreEmbarque] audit posta failed (non-blocking):', err);
      }
    }
    ```

There is NO `leadsService.appendPosta` method — do not invent one. The write goes through `deepCleanForFirestore` to honor `.claude/rules/firestore.md`.
    - `update(presupuestoId, ...)` already triggers the recompute hook plan 12-05 will add — so the cuota with hito='pre_embarque' transitions to `habilitada` automatically without an extra call here.

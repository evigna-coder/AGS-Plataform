```typescript
async generarAvisoFacturacion(
  presupuestoId: string,
  otNumbers: string[],
```

    A) Refactor `PresupuestoFacturacionSection.tsx` (target <=120 lines after refactor — pure orchestration):

    Replace the body so it only orchestrates the two sub-sections:

    ```typescript
    interface Props {
      presupuestoId: string;
      esquemaFacturacion: PresupuestoCuotaFacturacion[] | null | undefined;
      otsListasParaFacturar: string[];
      moneda: MonedaPresupuesto;
      itemsForTotals: PresupuestoItem[];
      onChanged: () => void;
      actor?: { uid: string; name?: string };
    }

    export function PresupuestoFacturacionSection(props: Props) {
      const esquema = props.esquemaFacturacion ?? [];
      const hasEsquema = esquema.length > 0;
      const allCuotasTerminal = hasEsquema && esquema.every(
        c => c.estado === 'facturada' || c.estado === 'cobrada'
      );
      // Sub-section B is visible when:
      // - There IS no esquema (legacy mode) AND there are OTs to factura
      // - OR there IS esquema, all cuotas are terminal, AND there are OTs left over (saldo)
      const showB = props.otsListasParaFacturar.length > 0 && (!hasEsquema || allCuotasTerminal);

      return (
        <section>
          {hasEsquema && (
            <CuotasDelEsquemaSection
              presupuestoId={props.presupuestoId}
              esquema={esquema}
              moneda={props.moneda}
              itemsForTotals={props.itemsForTotals}
              otsListasParaFacturar={props.otsListasParaFacturar}
              onGenerated={() => props.onChanged()}
              actor={props.actor}
            />
          )}
          {showB && (
            <OtsSinAsociarSection
              presupuestoId={props.presupuestoId}
              otsListasParaFacturar={props.otsListasParaFacturar}
              currency={props.moneda}
              onGenerated={() => props.onChanged()}
              actor={props.actor}
            />
          )}
          {!hasEsquema && props.otsListasParaFacturar.length === 0 && (
            <p className="text-xs uppercase tracking-wide font-mono text-slate-500">
              Sin actividad de facturación.
            </p>
          )}
        </section>
      );
    }
    ```

    Remove the entire legacy body that lived in this file (now in `OtsSinAsociarSection`). Preserve the export name and default-export shape so `EditPresupuestoModal.tsx` import still works.

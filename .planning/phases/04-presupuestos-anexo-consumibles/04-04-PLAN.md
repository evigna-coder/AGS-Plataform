    ```typescript
    function itemRequiereAnexo(item: PresupuestoItem, plantillas: TipoEquipoPlantilla[]): boolean {
      if (!item.servicioCode) return false;
      // Buscar el servicio en TODAS las plantillas — el flag puede estar en cualquiera
      // que contenga este servicioCode.
      for (const p of plantillas) {
        for (const s of p.servicios) {
          if (s.servicioCode === item.servicioCode) {
            return s.requiereAnexoConsumibles === true;   // estricto — undefined/false → no
          }
        }
      }
      return false;
    }
    ```
    NOTA: Un mismo `servicioCode` puede aparecer en varias plantillas (ej: `MP1_CN_60` en HPLC 1260 y en HPLC 1260 Infinity). La regla operativa: el flag se respeta si CUALQUIER plantilla lo marca. Esto previene falsos negativos cuando el operador tildó el flag en una plantilla pero no en otra equivalente. Documentar esta regla en JSDoc de la función.

    4. Función principal:
    ```typescript
    export async function buildAnexosFromPresupuesto(
      input: BuildAnexosInput
    ): Promise<{ anexos: AnexoBuildResult[]; warnings: AnexoBuildWarning[] }> {
      const { presupuesto, cliente, sistemas, modulosBySistema, plantillas } = input;
      const anexos: AnexoBuildResult[] = [];
      const warnings: AnexoBuildWarning[] = [];

      // Cache lookups por código para no hacer round-trips repetidos (mismo G7129A en 2 items distintos).
      const lookupCache = new Map<string, ConsumibleModulo[] | null>();
      // null sentinel = lookup hecho, no encontrado (caso ii). undefined = no consultado todavía.

      const lookup = async (codigo: string): Promise<ConsumibleModulo[] | null> => {
        if (lookupCache.has(codigo)) return lookupCache.get(codigo)!;
        const doc = await consumiblesPorModuloService.getByCodigoModulo(codigo);
        const result = doc && doc.activo ? doc.consumibles : null;
        lookupCache.set(codigo, result);
        return result;
      };

      const fechaEmision = (presupuesto.fechaEmision || presupuesto.createdAt || new Date().toISOString()) as string;

      for (const item of presupuesto.items) {
        if (!itemRequiereAnexo(item, plantillas)) continue;
        if (!item.sistemaId) {
          warnings.push({
            tipo: 'sistema_sin_modulos_ni_plantilla',
            itemId: item.id,
            sistemaNombre: item.sistemaNombre ?? null,
            detalle: 'El item lleva flag de anexo pero no tiene sistemaId vinculado',
          });
          continue;
        }

        const sistema = sistemas[item.sistemaId] ?? null;
        const sistemaNombre = item.sistemaNombre || sistema?.nombre || 'Sistema desconocido';

        // 1) Determinar fuente de módulos: reales > plantilla
        const modulosReales = modulosBySistema[item.sistemaId] || [];
        let fuente: { codigo: string; descripcion: string }[] = [];

        if (modulosReales.length > 0) {
          // Cada módulo real — derivar codigo/descripcion. NO tienen `codigo` standard;
          // usamos `marca`+`nombre` como heurística cuando falta.
          // CONTEXT.md decisión: caso (i) — "módulo sin código" se marca con placeholder.
          // Heurística: si el módulo tiene algo que parezca part number (campo `descripcion` que arranca con G/letra-número)
          //  o un campo explícito... NO hay tal campo en ModuloSistema actual.
          // Decisión: para la rama "módulos reales", asumimos que el campo de código vive en `nombre`
          // SOLO si matchea regex /^[A-Z][0-9]+[A-Z]?/ (part number Agilent). Si no, placeholder.
          for (const mod of modulosReales) {
            const candidato = (mod.nombre || '').trim();
            const isPartNumber = /^[A-Z][0-9]{3,5}[A-Z]?$/.test(candidato);
            if (isPartNumber) {
              fuente.push({ codigo: candidato, descripcion: mod.descripcion || mod.nombre });
            } else {
              fuente.push({ codigo: '', descripcion: mod.descripcion || mod.nombre || 'Módulo sin código' });
            }
          }
        } else {
          // 2) Fallback: plantilla
          const plantilla = sistema ? findPlantillaForSistema(sistema, plantillas) : null;
          if (!plantilla) {
            warnings.push({
              tipo: 'sistema_sin_modulos_ni_plantilla',
              itemId: item.id,
              sistemaNombre,
              detalle: `No se encontraron módulos reales del sistema "${sistemaNombre}" ni una plantilla matcheable. ¿Enviar sin anexo?`,
            });
            continue;
          }
          fuente = plantilla.componentes
            .slice()
            .sort((a, b) => a.orden - b.orden)
            .map(c => ({ codigo: c.codigo, descripcion: c.descripcion }));
        }

        // 3) Para cada fuente, lookup en consumiblesPorModulo + clasificar
        const modulosOut: AnexoModuloEntry[] = [];
        for (const { codigo, descripcion } of fuente) {
          if (!codigo) {
            // Caso (i): módulo sin código
            modulosOut.push({ codigo: '', descripcion, consumibles: [], placeholder: true });
            warnings.push({
              tipo: 'modulo_sin_codigo',
              itemId: item.id,
              sistemaNombre,
              detalle: `Módulo "${descripcion}" del sistema "${sistemaNombre}" no tiene código — listado sin consumibles específicos`,
            });
            continue;
          }
          const consumibles = await lookup(codigo);
          if (consumibles === null) {
            // Caso (ii): código no está en catálogo
            modulosOut.push({ codigo, descripcion, consumibles: [], placeholder: true });
            warnings.push({
              tipo: 'codigo_no_en_catalogo',
              itemId: item.id,
              sistemaNombre,
              detalle: `Código ${codigo} no está en consumibles_por_modulo`,
            });
            continue;
          }
          if (consumibles.length === 0) {
            // Caso (iii): catálogo intencionalmente vacío → SKIP silencioso
            continue;
          }
          // Caso happy: incluir
          modulosOut.push({ codigo, descripcion, consumibles, placeholder: false });
        }

        if (modulosOut.length === 0) {
          // Después de todos los skips silenciosos no quedó nada que mostrar.
          // Igual no es un error duro — el item lleva flag pero sus módulos están todos
          // marcados como "sin consumibles". Skip silencioso del anexo entero.
          continue;
        }

        const filename = `Anexo Consumibles - ${presupuesto.numero || presupuesto.id} - ${sistemaNombre}.pdf`;
        anexos.push({
          itemId: item.id,
          sistemaId: item.sistemaId,
          data: {
            presupuestoNumero: presupuesto.numero || presupuesto.id,
            sistemaNombre,
            clienteRazonSocial: cliente?.razonSocial || 'Cliente sin nombre',
            fechaEmision,
            modulos: modulosOut,
          },
          filename,
        });
      }

      return { anexos, warnings };
    }
    ```

```tsx
import { useEffect, useState } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { facturacionService } from '../../services/facturacionService';
import { useAuth } from '../../contexts/AuthContext';
import { PageHeader } from '../../components/ui/PageHeader';
import { Card } from '../../components/ui/Card';
import { Button } from '../../components/ui/Button';
import { useConfirm } from '../../components/ui/ConfirmDialog';
import type { SolicitudFacturacion } from '@ags/shared';
import { SOLICITUD_FACTURACION_ESTADO_LABELS, SOLICITUD_FACTURACION_ESTADO_COLORS, MONEDA_SIMBOLO } from '@ags/shared';

export const FacturacionDetail = () => {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const { firebaseUser, usuario, hasRole } = useAuth();
  const confirm = useConfirm();
  const [sol, setSol] = useState<SolicitudFacturacion | null>(null);
  const [loading, setLoading] = useState(true);
  const [notaDraft, setNotaDraft] = useState('');
  const [savingNota, setSavingNota] = useState(false);

  const canAdminAction = hasRole('admin', 'admin_soporte');
  const actor = { uid: firebaseUser?.uid || '', name: usuario?.nombre || undefined };

  useEffect(() => {
    if (!id) return;
    facturacionService.getById(id).then(s => {
      setSol(s);
      setNotaDraft(s?.observaciones || '');
      setLoading(false);
    }).catch(() => setLoading(false));
  }, [id]);

  const reload = async () => {
    if (!id) return;
    const fresh = await facturacionService.getById(id);
    setSol(fresh);
    setNotaDraft(fresh?.observaciones || '');
  };

  const handleMarcarEnviada = async () => {
    if (!sol) return;
    if (!await confirm(`¿Marcar ${sol.presupuestoNumero} como enviada al contable?`)) return;
    await facturacionService.marcarEnviada(sol.id, actor);
    await reload();
  };

  const handleMarcarFacturada = async () => {
    if (!sol) return;
    if (!await confirm(`¿Marcar ${sol.presupuestoNumero} como facturada?`)) return;
    await facturacionService.marcarFacturada(sol.id, actor);
    await reload();
  };

  const handleSaveNota = async () => {
    if (!sol) return;
    setSavingNota(true);
    await facturacionService.agregarNota(sol.id, notaDraft, actor);
    await reload();
    setSavingNota(false);
  };

  if (loading) return <p className="text-center py-8 text-sm text-slate-400">Cargando...</p>;
  if (!sol) return <p className="text-center py-8 text-sm text-slate-400">Solicitud no encontrada</p>;

  return (
    <div className="space-y-4">
      <PageHeader
        title={`Solicitud ${sol.presupuestoNumero}`}
        subtitle={sol.clienteNombre}
        onBack={() => navigate('/facturacion')}
      />

      <Card>
        <div className="grid grid-cols-2 gap-4 text-xs">
          <div><span className="font-mono uppercase text-[10px] text-slate-400">Estado</span>
            <span className={`inline-block ml-2 px-2 py-0.5 rounded-full text-[10px] font-medium ${SOLICITUD_FACTURACION_ESTADO_COLORS[sol.estado]}`}>
              {SOLICITUD_FACTURACION_ESTADO_LABELS[sol.estado]}
            </span>
          </div>
          <div><span className="font-mono uppercase text-[10px] text-slate-400">Monto</span>
            <span className="ml-2 font-mono">
              {MONEDA_SIMBOLO[sol.moneda] || '$'} {sol.montoTotal.toLocaleString('es-AR', { minimumFractionDigits: 2 })}
            </span>
          </div>
          {sol.otNumbers && sol.otNumbers.length > 0 && (
            <div><span className="font-mono uppercase text-[10px] text-slate-400">OTs vinculadas</span>
              <span className="ml-2">{sol.otNumbers.join(', ')}</span>
            </div>
          )}
          {sol.ordenesCompraIds && sol.ordenesCompraIds.length > 0 && (
            <div><span className="font-mono uppercase text-[10px] text-slate-400">OCs vinculadas</span>
              <span className="ml-2">{sol.ordenesCompraIds.length} OC(s)</span>
            </div>
          )}
          {sol.numeroFactura && (
            <div><span className="font-mono uppercase text-[10px] text-slate-400">N° Factura</span>
              <span className="ml-2 font-mono">{sol.numeroFactura}</span>
            </div>
          )}
        </div>
      </Card>

      {/* Items snapshot */}
      <Card>
        <h3 className="text-sm font-semibold mb-2">Items del presupuesto</h3>
        <table className="w-full text-xs">
          <thead>
            <tr className="border-b border-slate-200">
              <th className="text-left py-1">Descripción</th>
              <th className="text-right py-1 w-20">Cant.</th>
              <th className="text-right py-1 w-24">P. Unit</th>
              <th className="text-right py-1 w-28">Subtotal</th>
            </tr>
          </thead>
          <tbody>
            {sol.items.map(it => (
              <tr key={it.id} className="border-b border-slate-100">
                <td className="py-1">{it.descripcion}</td>
                <td className="text-right">{it.cantidad}</td>
                <td className="text-right font-mono">{it.precioUnitario.toLocaleString('es-AR', { minimumFractionDigits: 2 })}</td>
                <td className="text-right font-mono">{it.subtotal.toLocaleString('es-AR', { minimumFractionDigits: 2 })}</td>
              </tr>
            ))}
          </tbody>
        </table>
      </Card>

      {/* Nota del contable */}
      <Card>
        <h3 className="text-sm font-semibold mb-2">Nota del contable</h3>
        {canAdminAction ? (
          <>
            <textarea
              value={notaDraft}
              onChange={e => setNotaDraft(e.target.value)}
              rows={3}
              className="w-full border border-slate-200 rounded-md px-2 py-1.5 text-xs"
              placeholder="Observaciones..."
            />
            <div className="flex gap-2 mt-2">
              <Button size="sm" onClick={handleSaveNota} disabled={savingNota || notaDraft === (sol.observaciones || '')}>
                {savingNota ? 'Guardando...' : 'Guardar nota'}
              </Button>
            </div>
          </>
        ) : (
          <p className="text-xs text-slate-600 whitespace-pre-wrap">{sol.observaciones || '—'}</p>
        )}
      </Card>

      {/* Acciones admin */}
      {canAdminAction && (
        <Card>
          <h3 className="text-sm font-semibold mb-2">Acciones</h3>
          <div className="flex gap-2">
            {sol.estado === 'pendiente' && (
              <Button variant="primary" size="sm" onClick={handleMarcarEnviada}>
                Marcar enviada
              </Button>
            )}
            {(sol.estado === 'pendiente' || sol.estado === 'enviada') && (
              <Button variant="primary" size="sm" onClick={handleMarcarFacturada}>
                Marcar facturada
              </Button>
            )}
          </div>
        </Card>
      )}
    </div>
  );
};
```

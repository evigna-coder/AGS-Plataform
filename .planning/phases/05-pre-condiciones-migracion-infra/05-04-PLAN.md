    ```typescript
    import { doc, getDoc, setDoc, onSnapshot, Timestamp } from 'firebase/firestore';
    import { db, deepCleanForFirestore } from './firebase';

    export interface ModuleFlag {
      enabled: boolean;
      updatedAt: string;  // ISO (se convierte al leer)
      updatedBy: string;
    }

    export interface FeatureFlagsModules {
      modules: Record<string, ModuleFlag>;
    }

    const DOC_PATH = ['featureFlags', 'modules'] as const;

    /**
     * Suscribe a cambios del doc /featureFlags/modules. El callback recibe el estado current
     * (o `{ modules: {} }` si el doc no existe). Devuelve la función de unsubscribe.
     */
    export function subscribeFeatureFlags(cb: (flags: FeatureFlagsModules) => void): () => void {
      const ref = doc(db, DOC_PATH[0], DOC_PATH[1]);
      return onSnapshot(ref, snap => {
        if (!snap.exists()) {
          cb({ modules: {} });
          return;
        }
        const data = snap.data();
        // Normalize timestamps → ISO
        const modules: Record<string, ModuleFlag> = {};
        if (data && typeof data.modules === 'object' && data.modules !== null) {
          for (const [path, raw] of Object.entries<any>(data.modules)) {
            modules[path] = {
              enabled: !!raw.enabled,
              updatedAt: raw.updatedAt?.toDate?.()?.toISOString() ?? '',
              updatedBy: raw.updatedBy ?? '',
            };
          }
        }
        cb({ modules });
      });
    }

    /** Lee el doc una vez (no-reactive). Útil para UI inicial. */
    export async function getFeatureFlagsOnce(): Promise<FeatureFlagsModules> {
      const ref = doc(db, DOC_PATH[0], DOC_PATH[1]);
      const snap = await getDoc(ref);
      if (!snap.exists()) return { modules: {} };
      const data = snap.data();
      const modules: Record<string, ModuleFlag> = {};
      for (const [path, raw] of Object.entries<any>(data.modules ?? {})) {
        modules[path] = {
          enabled: !!raw.enabled,
          updatedAt: raw.updatedAt?.toDate?.()?.toISOString() ?? '',
          updatedBy: raw.updatedBy ?? '',
        };
      }
      return { modules };
    }

    /**
     * Togglea un módulo por su path. Usa setDoc con merge para no pisar otros módulos.
     * Si el doc no existe, lo crea con este único módulo.
     */
    export async function setModuleEnabled(modulePath: string, enabled: boolean, uid: string): Promise<void> {
      const ref = doc(db, DOC_PATH[0], DOC_PATH[1]);
      const now = Timestamp.now();
      const payload = deepCleanForFirestore({
        modules: {
          [modulePath]: { enabled, updatedAt: now, updatedBy: uid },
        },
        updatedAt: now,
      });
      await setDoc(ref, payload, { merge: true });
    }
    ```

    import { createContext, useContext, useEffect, useState, type ReactNode } from 'react';
    import { subscribeFeatureFlags, type FeatureFlagsModules } from '../services/featureFlagsService';

    const FeatureFlagsContext = createContext<FeatureFlagsModules | null>(null);

    /**
     * Provider que se suscribe al doc /featureFlags/modules al mount y actualiza el
     * contexto en cada snapshot. El `null` inicial indica "cargando"; tras el primer
     * snapshot (que puede ser `{ modules: {} }` si no existe el doc) se vuelve non-null.
     */
    export function FeatureFlagsProvider({ children }: { children: ReactNode }) {
      const [flags, setFlags] = useState<FeatureFlagsModules | null>(null);

      useEffect(() => {
        const unsub = subscribeFeatureFlags(next => setFlags(next));
        return () => unsub();
      }, []);

      return (
        <FeatureFlagsContext.Provider value={flags}>
          {children}
        </FeatureFlagsContext.Provider>
      );
    }

    /** Hook. Devuelve `null` mientras carga el primer snapshot; después `{ modules: {...} }`. */
    export function useFeatureFlags(): FeatureFlagsModules | null {
      return useContext(FeatureFlagsContext);
    }

              const defaultEnabled = isMvpDefault(mod.path);

                  defaultEnabled={defaultEnabled}

                />
              );
            })}
          </Card>
        </div>
      );
    }

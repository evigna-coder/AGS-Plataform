```typescript
import { useState } from 'react';
import { Timestamp } from 'firebase/firestore';
import { Importacion, ItemImportacion } from '@ags/shared';
import { createBatch, deepCleanForFirestore, getUpdateTrace } from '../services/firebase';
import { docRef } from '../services/firebase';  // verify exact import path in project
import { calcularCostoConGastos } from '../utils/calcularProrrateo';

export interface RecepcionItem {
  item: ItemImportacion;
  posicionId: string;
  posicionNombre: string;
  nrosSerie: string[];     // one per unit; empty array if no serial required
  cantidadReal: number;
}

export function useIngresarStock() {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const ingresarStock = async (imp: Importacion, recepciones: RecepcionItem[]) => {
    setLoading(true);
    setError(null);
    try {
      // 1. Compute prorrateo base values
      const monedaOC = imp.moneda ?? 'USD';
      const totalGastosEnMonedaOC = (imp.gastos ?? [])
        .filter(g => g.moneda === monedaOC)
        .reduce((sum, g) => sum + g.monto, 0);

      const valorTotalImportacion = recepciones.reduce(
        (sum, r) => sum + (r.item.precioUnitario ?? 0) * r.cantidadReal, 0
      );

      // 2. Build atomic batch
      const batch = createBatch();

      for (const rec of recepciones) {
        const costoUnitario = calcularCostoConGastos({
          precioUnitario: rec.item.precioUnitario ?? 0,
          cantidadRecibida: rec.cantidadReal,
          valorTotalImportacion,
          totalGastosEnMonedaOC,
        });

        // One UnidadStock per serial (or one if no serials)
        const seriesONull: (string | null)[] = rec.nrosSerie.length > 0
          ? rec.nrosSerie
          : [null];

        for (const nroSerie of seriesONull) {
          const unidadId = crypto.randomUUID();
          const movId = crypto.randomUUID();

          const unidad = deepCleanForFirestore({
            articuloId: rec.item.articuloId ?? '',
            articuloCodigo: rec.item.articuloCodigo ?? null,
            descripcion: rec.item.descripcion,
            estado: 'disponible' as const,
            ubicacion: rec.posicionNombre,
            posicionStockId: rec.posicionId,
            nroSerie: nroSerie,
            costoUnitario,
            monedaCosto: monedaOC,
            importacionId: imp.id,
            createdAt: Timestamp.now(),
            updatedAt: Timestamp.now(),
            ...getUpdateTrace(),
          });

          batch.set(docRef('unidades_stock', unidadId), unidad);

          const movimiento = deepCleanForFirestore({
            tipo: 'ingreso' as const,
            unidadStockId: unidadId,
            articuloId: rec.item.articuloId ?? '',
            cantidad: 1,
            origenTipo: 'proveedor' as const,
            origenId: imp.id,
            destinoTipo: 'posicion' as const,
            destinoId: rec.posicionId,
            nota: `Ingreso por importación #${imp.numero}`,
            createdAt: Timestamp.now(),
            ...getUpdateTrace(),
          });

          batch.set(docRef('movimientos_stock', movId), movimiento);
        }

        // Close requirement if quantity fulfilled
        if (rec.item.requerimientoId && rec.cantidadReal >= rec.item.cantidadPedida) {
          batch.update(docRef('requerimientos_compra', rec.item.requerimientoId), {
            estado: 'completado',
            updatedAt: Timestamp.now(),
            ...getUpdateTrace(),
          });
        }
      }

      // Mark importacion as done
      batch.update(docRef('importaciones', imp.id), deepCleanForFirestore({
        stockIngresado: true,
        items: recepciones.map(r => ({
          ...r.item,
          cantidadRecibida: r.cantidadReal,
        })),
        updatedAt: Timestamp.now(),
        ...getUpdateTrace(),
      }));

      await batch.commit();
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : 'Error al ingresar stock');
    } finally {
      setLoading(false);
    }
  };

  return { ingresarStock, loading, error };
}
```
